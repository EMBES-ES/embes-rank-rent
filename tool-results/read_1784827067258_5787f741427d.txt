    12→function getClusterServices(service: Service): Service[] {
    13→  const cluster = clusters.find((c) => c.id === service.cluster);
    14→  return cluster ? cluster.services : [];
    15→}

    17→/** Deterministic pseudo-variation based on slug chars to add subtle uniqueness */
    18→function locVariation(location: Location): number {
    19→  let h = 0;
    20→  for (let i = 0; i < location.slug.length; i++) {
    21→    h = (h * 31 + location.slug.charCodeAt(i)) | 0;
    22→  }
    23→  return Math.abs(h);
    24→}
    25→
    26→/** Pick from an array deterministically based on a seed */
    27→function pick<T>(arr: T[], seed: number): T {
    28→  return arr[seed % arr.length];
    29→}

   291→function generateIntro(service: Service, location: Location): string {

   299→    .replace(/{s}/g, service.shortName.toLowerCase())
   300→    .replace(/{l}/g, location.name);
   301→

   303→
   304→  const cta = pick(ctas, v + 7)
   305→    .replace(/{l}/g, location.name)
   306→    .replace(/{c}/g, location.community)
   307→    .replace(/{p}/g, location.province);
   308→
   309→  return `${opening} ${service.description.charAt(0).toLowerCase()}${service.description.slice(1)} ${benefit} ${cta}`;
